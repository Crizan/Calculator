class test{

	int width,length,height,depth;

	test(test obj){
		width=obj.width;
		length=obj.length;
		height=obj.height;
		depth=100;
	}
	test(int width,int length,int height){
		this.width=width;
		this.length=length;
		this.height=height;
	}
	void display(){
		System.out.println(" Width: "+width+"\n length: "+length+"\n height: "+height+"\n depth: "+depth);
	}

}


class object_parameter{

	public static void main(String args[]){
		test obj1= new test(12,10,25);
		test obj2=new test(obj1);
		obj1.display();
	
	}
}
